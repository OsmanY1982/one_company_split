# -*- coding: utf-8 -*-
"""
火星 — 铁锈红+棕褐高原+极冠白斑+暗色峡谷纹理
"""
import math, random
from PyQt5.QtCore import Qt, QPointF
from PyQt5.QtGui import (
    QPainter, QRadialGradient, QColor, QPen, QBrush, QPainterPath
)


def paint(painter: QPainter, center: QPointF, radius: float,
          anim_t: float, hovered: bool, alpha: float):
    cx, cy = center.x(), center.y()
    p = painter
    p.save()
    if alpha < 1.0:
        p.setOpacity(alpha)

    # ── 外辉光 ──
    for gl in range(4):
        gr = radius * (1.06 + gl * 0.20)
        g = QRadialGradient(cx, cy, gr)
        ga = max(1, 35 - gl * 8)
        g.setColorAt(0.0, QColor(255, 255, 255, 0))
        g.setColorAt(0.25, QColor(255, 200, 150, ga // 2))
        g.setColorAt(0.55, QColor(220, 140, 80, ga))
        g.setColorAt(0.80, QColor(150, 80, 30, ga // 2))
        g.setColorAt(1.0, QColor(0, 0, 0, 0))
        p.setBrush(g); p.setPen(Qt.NoPen)
        p.drawEllipse(center, gr, gr)

    # ── 基底（铁锈红→暗棕褐渐变）──
    body = QRadialGradient(cx - radius * 0.10, cy - radius * 0.10, radius * 1.04)
    body.setColorAt(0.0, QColor(220, 130, 65))
    body.setColorAt(0.20, QColor(195, 105, 50))
    body.setColorAt(0.42, QColor(160, 80, 35))
    body.setColorAt(0.62, QColor(125, 55, 22))
    body.setColorAt(0.78, QColor(85, 35, 14))
    body.setColorAt(0.90, QColor(50, 18, 8))
    body.setColorAt(1.0, QColor(22, 8, 3))
    p.setBrush(body); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 暗色高原（类Syrtis Major 等暗色玄武岩区域）──
    highland_rng = random.Random(51)
    for _ in range(6):
        hx = cx + highland_rng.uniform(-radius * 0.50, radius * 0.50)
        hy = cy + highland_rng.uniform(-radius * 0.50, radius * 0.50)
        hr = radius * highland_rng.uniform(0.10, 0.22)
        if (hx - cx) ** 2 + (hy - cy) ** 2 > (radius * 0.75) ** 2:
            continue
        hg = QRadialGradient(hx, hy, hr)
        ha = highland_rng.randint(40, 90)
        hg.setColorAt(0.0, QColor(70, 30, 12, ha))
        hg.setColorAt(0.5, QColor(100, 50, 20, ha // 2))
        hg.setColorAt(1.0, QColor(255, 255, 255, 0))
        p.setBrush(hg); p.setPen(Qt.NoPen)
        p.drawEllipse(QPointF(hx, hy), hr, hr)

    # ── 峡谷纹理（类Valles Marineris）──
    canyon_path = QPainterPath()
    cx0, cy0 = cx - radius * 0.05, cy - radius * 0.02
    canyon_path.moveTo(cx0, cy0)
    canyon_dir = -0.3
    for i in range(15):
        frac = i / 14.0
        nx = cx0 + math.cos(canyon_dir + frac * 0.8) * radius * 0.42 * frac
        ny = cy0 + math.sin(canyon_dir + frac * 0.8) * radius * 0.18 * frac
        canyon_path.lineTo(nx, ny)
    canyon_pen = QPen(QColor(40, 15, 8, 90), radius * 0.035)
    canyon_pen.setCapStyle(Qt.RoundCap)
    p.setPen(canyon_pen); p.setBrush(Qt.NoBrush)
    p.drawPath(canyon_path)
    # 峡谷内高亮线
    canyon_pen2 = QPen(QColor(180, 100, 45, 40), radius * 0.008)
    canyon_pen2.setCapStyle(Qt.RoundCap)
    p.setPen(canyon_pen2)
    p.drawPath(canyon_path)

    # ── 地表沙尘纹理（随机斑点模拟风蚀地貌）──
    dust_rng = random.Random(67)
    for _ in range(30):
        dx = cx + dust_rng.uniform(-radius * 0.78, radius * 0.78)
        dy = cy + dust_rng.uniform(-radius * 0.78, radius * 0.78)
        if (dx - cx) ** 2 + (dy - cy) ** 2 > (radius * 0.88) ** 2:
            continue
        dr = radius * dust_rng.uniform(0.02, 0.07)
        dg = QRadialGradient(dx, dy, dr)
        da = dust_rng.randint(25, 70)
        dc = dust_rng.choice([QColor(200, 120, 55, da), QColor(130, 65, 25, da),
                               QColor(170, 90, 35, da)])
        dg.setColorAt(0.0, dc)
        dg.setColorAt(1.0, QColor(255, 255, 255, 0))
        p.setBrush(dg); p.setPen(Qt.NoPen)
        p.drawEllipse(QPointF(dx, dy), dr, dr)

    # ── 极冠白斑（北极冰盖+南极小冰盖）──
    # 北极冠
    npole = QRadialGradient(cx, cy - radius * 0.68, radius * 0.38)
    npole.setColorAt(0.0, QColor(240, 235, 225, 160))
    npole.setColorAt(0.25, QColor(230, 220, 210, 110))
    npole.setColorAt(0.5, QColor(210, 195, 180, 50))
    npole.setColorAt(0.75, QColor(200, 180, 160, 15))
    npole.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(npole); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)
    # 南极冠（更小）
    spole = QRadialGradient(cx, cy + radius * 0.68, radius * 0.25)
    spole.setColorAt(0.0, QColor(235, 230, 220, 130))
    spole.setColorAt(0.4, QColor(220, 210, 200, 60))
    spole.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(spole); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 稀薄大气光晕（极淡的粉/橙色边缘）──
    atmo = QRadialGradient(cx, cy, radius * 1.07)
    atmo.setColorAt(0.0, QColor(255, 255, 255, 0))
    atmo.setColorAt(0.75, QColor(255, 255, 255, 0))
    atmo.setColorAt(0.88, QColor(255, 180, 120, 22))
    atmo.setColorAt(0.95, QColor(220, 140, 80, 15))
    atmo.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(atmo); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius * 1.07, radius * 1.07)

    # ── 高光 ──
    hl = QRadialGradient(cx - radius * 0.24, cy - radius * 0.26, radius * 0.30)
    hl.setColorAt(0.0, QColor(255, 220, 170, 50))
    hl.setColorAt(0.5, QColor(240, 180, 110, 15))
    hl.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(hl); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 暗面 ──
    shadow = QRadialGradient(cx + radius * 0.50, cy + radius * 0.42, radius * 0.78)
    shadow.setColorAt(0.0, QColor(255, 255, 255, 0))
    shadow.setColorAt(0.30, QColor(0, 0, 0, 30))
    shadow.setColorAt(0.52, QColor(0, 0, 0, 85))
    shadow.setColorAt(0.73, QColor(0, 0, 0, 160))
    shadow.setColorAt(0.91, QColor(0, 0, 0, 220))
    shadow.setColorAt(1.0, QColor(0, 0, 0, 240))
    p.setBrush(shadow); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 边缘逆光 ──
    rim_breath = 1.0 + 0.03 * math.sin(anim_t * 2.2)
    rim = QRadialGradient(cx + radius * 0.45, cy + radius * 0.50, radius * 0.50)
    rim.setColorAt(0.0, QColor(255, 255, 255, 0))
    rim.setColorAt(0.55, QColor(255, 255, 255, 0))
    rim.setColorAt(0.78, QColor(255, 180, 130, int(15 * rim_breath)))
    rim.setColorAt(0.92, QColor(220, 140, 90, int(30 * rim_breath)))
    rim.setColorAt(1.0, QColor(0, 0, 0, 0))
    p.setBrush(rim); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 悬停增强 ──
    if hovered:
        hp = 0.7 + 0.3 * abs(math.sin(anim_t * 3.5))
        for i in range(3):
            ir = radius + 2 + i * 5
            ig = QRadialGradient(center, ir)
            ga = int((70 - i * 18) * hp)
            ig.setColorAt(0.60, QColor(255, 255, 255, 0))
            ig.setColorAt(0.78, QColor(220, 130, 60, ga // 2))
            ig.setColorAt(0.90, QColor(220, 130, 60, ga))
            ig.setColorAt(0.97, QColor(110, 65, 40, ga // 3))
            ig.setColorAt(1.0, QColor(0, 0, 0, 0))
            p.setPen(Qt.NoPen); p.setBrush(ig)
            p.drawEllipse(center, ir, ir)
        for i in range(3):
            outer_r = radius + 10 + i * 10
            og = QRadialGradient(center, outer_r)
            ga = int((50 - i * 14) * hp)
            og.setColorAt(0.75, QColor(255, 255, 255, 0))
            og.setColorAt(0.88, QColor(220, 130, 60, ga // 2))
            og.setColorAt(0.96, QColor(110, 65, 50, ga // 3))
            og.setColorAt(1.0, QColor(0, 0, 0, 0))
            p.setPen(Qt.NoPen); p.setBrush(og)
            p.drawEllipse(center, outer_r, outer_r)
        br = 0.6 + 0.4 * abs(math.sin(anim_t * 4.0))
        rpen = QPen(QColor(220, 130, 60, int(220 * hp * br)), 2.5 + 1.0 * br)
        p.setPen(rpen); p.setBrush(Qt.NoBrush)
        p.drawEllipse(center, radius + 3, radius + 3)

    p.restore()
