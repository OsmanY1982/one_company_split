# -*- coding: utf-8 -*-
"""
气态巨行星 — 橙棕黄多层水平条纹+大红斑+色带扰动
"""
import math, random
from PyQt5.QtCore import Qt, QPointF
from PyQt5.QtGui import (
    QPainter, QRadialGradient, QColor, QPen, QBrush, QPainterPath
)


def band_color(t, base_r, base_g, base_b):
    """根据位置 t (0~1) 返回扰动后的颜色"""
    r = base_r + int(25 * math.sin(t * 15 + 3.3))
    g = base_g + int(20 * math.cos(t * 18 + 1.7))
    b = base_b + int(12 * math.sin(t * 20))
    return QColor(min(255, max(0, r)), min(255, max(0, g)), min(255, max(0, b)))


def paint(painter: QPainter, center: QPointF, radius: float,
          anim_t: float, hovered: bool, alpha: float):
    cx, cy = center.x(), center.y()
    p = painter
    p.save()
    if alpha < 1.0:
        p.setOpacity(alpha)

    # ── 外辉光 ──
    for gl in range(4):
        gr = radius * (1.06 + gl * 0.20)
        g = QRadialGradient(cx, cy, gr)
        ga = max(1, 35 - gl * 8)
        g.setColorAt(0.0, QColor(255, 255, 255, 0))
        g.setColorAt(0.25, QColor(200, 200, 255, ga // 2))
        g.setColorAt(0.55, QColor(120, 140, 255, ga))
        g.setColorAt(0.80, QColor(60, 80, 200, ga // 2))
        g.setColorAt(1.0, QColor(0, 0, 0, 0))
        p.setBrush(g); p.setPen(Qt.NoPen)
        p.drawEllipse(center, gr, gr)

    # ── 球体基底（暖橙→棕渐变）──
    base = QRadialGradient(cx - radius * 0.12, cy - radius * 0.10, radius * 1.03)
    base.setColorAt(0.0, QColor(250, 225, 165))
    base.setColorAt(0.25, QColor(225, 190, 130))
    base.setColorAt(0.50, QColor(195, 155, 100))
    base.setColorAt(0.72, QColor(155, 115, 70))
    base.setColorAt(0.88, QColor(110, 75, 45))
    base.setColorAt(1.0, QColor(65, 40, 22))
    p.setBrush(base); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 水平条纹带（多层，带纵向扰动模拟大气湍流）──
    band_defs = [
        (0.92, QColor(180, 110, 55), 80),
        (0.84, QColor(240, 180, 100), 60),
        (0.76, QColor(160, 95, 45), 90),
        (0.68, QColor(250, 200, 120), 50),
        (0.58, QColor(175, 105, 50), 85),
        (0.48, QColor(255, 210, 140), 55),
        (0.38, QColor(155, 90, 40), 95),
        (0.28, QColor(245, 195, 115), 65),
        (0.18, QColor(165, 100, 48), 80),
        (0.08, QColor(235, 185, 105), 55),
        (-0.06, QColor(150, 88, 38), 90),
        (-0.18, QColor(245, 200, 125), 60),
        (-0.28, QColor(160, 95, 42), 85),
        (-0.40, QColor(240, 190, 110), 55),
        (-0.52, QColor(170, 102, 46), 80),
        (-0.66, QColor(250, 215, 135), 50),
        (-0.80, QColor(155, 90, 38), 88),
        (-0.92, QColor(175, 108, 52), 75),
    ]

    for rel_y, col, band_alpha in band_defs:
        band_cy = cy + rel_y * radius
        # 扰动：波段有波状偏移
        wave = math.sin(anim_t * 0.5 + rel_y * 12) * radius * 0.03
        band_cy += wave

        dx = radius * math.sqrt(max(0, 1 - ((band_cy - cy) / radius) ** 2))
        if dx <= 0:
            continue
        # 波动条纹路径
        bp = QPainterPath()
        segs = 20
        for s in range(segs + 1):
            frac = s / segs
            bx = cx - dx + 2 * dx * frac
            by_wave = band_cy + math.sin(frac * 18 + anim_t * 1.3 + rel_y * 9) * radius * 0.015
            if s == 0:
                bp.moveTo(bx, by_wave)
            else:
                bp.lineTo(bx, by_wave)
        pen = QPen(QColor(col.red(), col.green(), col.blue(), band_alpha), radius * 0.06)
        pen.setCapStyle(Qt.RoundCap)
        p.setPen(pen); p.setBrush(Qt.NoBrush)
        p.drawPath(bp)

    # ── 大红斑（椭圆形巨型风暴，赤道南侧）──
    spot_cx = cx + radius * 0.28
    spot_cy = cy + radius * 0.12
    spot_rx = radius * 0.16
    spot_ry = radius * 0.10
    spot_rot = anim_t * 0.4
    p.save()
    p.translate(spot_cx, spot_cy)
    p.rotate(math.degrees(spot_rot * 0.1))
    p.scale(1.0, 0.65)
    # 斑核
    spot_core = QRadialGradient(0, 0, spot_rx * 1.3)
    spot_core.setColorAt(0.0, QColor(240, 130, 85, 200))
    spot_core.setColorAt(0.3, QColor(220, 100, 60, 180))
    spot_core.setColorAt(0.6, QColor(200, 80, 40, 140))
    spot_core.setColorAt(0.8, QColor(255, 140, 100, 60))
    spot_core.setColorAt(1.0, QColor(255, 180, 140, 10))
    p.setBrush(spot_core); p.setPen(Qt.NoPen)
    p.drawEllipse(QPointF(0, 0), spot_rx, spot_ry)
    # 斑眼
    spot_eye = QRadialGradient(-spot_rx * 0.05, -spot_ry * 0.05, spot_rx * 0.25)
    spot_eye.setColorAt(0.0, QColor(255, 200, 160, 120))
    spot_eye.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(spot_eye)
    p.drawEllipse(QPointF(0, 0), spot_rx, spot_ry)
    p.restore()

    # ── 涡旋扰动（中小风暴斑点）──
    storm_rng = random.Random(77)
    for _ in range(12):
        sx = cx + storm_rng.uniform(-radius * 0.55, radius * 0.55)
        sy = cy + storm_rng.uniform(-radius * 0.55, radius * 0.55)
        if (sx - cx) ** 2 + (sy - cy) ** 2 > (radius * 0.88) ** 2:
            continue
        ss = radius * storm_rng.uniform(0.02, 0.06)
        sg = QRadialGradient(sx, sy, ss * 1.5)
        sa = storm_rng.randint(30, 90)
        sc = storm_rng.choice([QColor(240, 160, 80, sa), QColor(180, 100, 50, sa),
                                QColor(255, 200, 130, sa)])
        sg.setColorAt(0.0, sc)
        sg.setColorAt(1.0, QColor(255, 255, 255, 0))
        p.setBrush(sg); p.setPen(Qt.NoPen)
        p.drawEllipse(QPointF(sx, sy), ss * 1.5, ss * 1.5)

    # ── 高光 ──
    hl = QRadialGradient(cx - radius * 0.22, cy - radius * 0.25, radius * 0.30)
    hl.setColorAt(0.0, QColor(255, 250, 220, 50))
    hl.setColorAt(0.5, QColor(255, 230, 180, 15))
    hl.setColorAt(1.0, QColor(255, 255, 255, 0))
    p.setBrush(hl); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 暗面 ──
    shadow = QRadialGradient(cx + radius * 0.55, cy + radius * 0.45, radius * 0.75)
    shadow.setColorAt(0.0, QColor(255, 255, 255, 0))
    shadow.setColorAt(0.30, QColor(0, 0, 0, 25))
    shadow.setColorAt(0.50, QColor(0, 0, 0, 75))
    shadow.setColorAt(0.72, QColor(0, 0, 0, 150))
    shadow.setColorAt(0.90, QColor(0, 0, 0, 210))
    shadow.setColorAt(1.0, QColor(0, 0, 0, 235))
    p.setBrush(shadow); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 边缘逆光 ──
    rim_breath = 1.0 + 0.03 * math.sin(anim_t * 2.2)
    rim = QRadialGradient(cx + radius * 0.45, cy + radius * 0.50, radius * 0.50)
    rim.setColorAt(0.0, QColor(255, 255, 255, 0))
    rim.setColorAt(0.55, QColor(255, 255, 255, 0))
    rim.setColorAt(0.78, QColor(200, 180, 255, int(15 * rim_breath)))
    rim.setColorAt(0.92, QColor(150, 130, 255, int(30 * rim_breath)))
    rim.setColorAt(1.0, QColor(0, 0, 0, 0))
    p.setBrush(rim); p.setPen(Qt.NoPen)
    p.drawEllipse(center, radius, radius)

    # ── 悬停增强 ──
    if hovered:
        hp = 0.7 + 0.3 * abs(math.sin(anim_t * 3.5))
        for i in range(3):
            ir = radius + 2 + i * 5
            ig = QRadialGradient(center, ir)
            ga = int((70 - i * 18) * hp)
            ig.setColorAt(0.60, QColor(255, 255, 255, 0))
            ig.setColorAt(0.78, QColor(240, 170, 80, ga // 2))
            ig.setColorAt(0.90, QColor(240, 170, 80, ga))
            ig.setColorAt(0.97, QColor(120, 85, 60, ga // 3))
            ig.setColorAt(1.0, QColor(0, 0, 0, 0))
            p.setPen(Qt.NoPen); p.setBrush(ig)
            p.drawEllipse(center, ir, ir)
        for i in range(3):
            outer_r = radius + 10 + i * 10
            og = QRadialGradient(center, outer_r)
            ga = int((50 - i * 14) * hp)
            og.setColorAt(0.75, QColor(255, 255, 255, 0))
            og.setColorAt(0.88, QColor(240, 170, 80, ga // 2))
            og.setColorAt(0.96, QColor(120, 85, 70, ga // 3))
            og.setColorAt(1.0, QColor(0, 0, 0, 0))
            p.setPen(Qt.NoPen); p.setBrush(og)
            p.drawEllipse(center, outer_r, outer_r)
        br = 0.6 + 0.4 * abs(math.sin(anim_t * 4.0))
        rpen = QPen(QColor(240, 170, 80, int(220 * hp * br)), 2.5 + 1.0 * br)
        p.setPen(rpen); p.setBrush(Qt.NoBrush)
        p.drawEllipse(center, radius + 3, radius + 3)

    p.restore()
