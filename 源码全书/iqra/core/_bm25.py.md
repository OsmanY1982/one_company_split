# -*- coding: utf-8 -*-
"""BM25 全文检索引擎 — 标题加权 + 同义词扩展 + 精确匹配 boost"""
import math
import re
from typing import List, Dict, Tuple, Optional
from collections import defaultdict

from ._tokenizer import Tokenizer

# ─── 代码搜索同义词映射 ─────────────────────────────────
_SYNONYM_MAP: Dict[str, List[str]] = {
    "add": ["create", "insert", "append", "push"],
    "create": ["add", "new", "make", "build", "generate"],
    "delete": ["remove", "destroy", "drop", "clear", "erase"],
    "remove": ["delete", "clear", "pop", "strip", "purge"],
    "update": ["modify", "change", "edit", "refresh", "patch"],
    "get": ["fetch", "retrieve", "find", "query", "read", "load"],
    "set": ["assign", "write", "put", "store", "save"],
    "error": ["exception", "fail", "crash", "bug", "fault"],
    "config": ["configuration", "settings", "options", "preference"],
    "auth": ["authentication", "login", "credential", "token"],
    "user": ["account", "member", "profile", "customer"],
    "file": ["document", "asset", "resource", "attachment"],
    "search": ["find", "query", "lookup", "scan", "filter"],
}
# 规范化小写映射（预展开）
_EXPANDED_SYNONYMS: Dict[str, set] = {}
for k, v in _SYNONYM_MAP.items():
    _EXPANDED_SYNONYMS[k.lower()] = {x.lower() for x in v}
    for syn in v:
        _EXPANDED_SYNONYMS.setdefault(syn.lower(), set()).add(k.lower())


class BM25:
    """BM25 全文检索（标题加权 + 同义词扩展）"""

    def __init__(self, k1: float = 1.5, b: float = 0.75,
                 title_lines: int = 8, title_boost: float = 2.0):
        """
        Args:
            k1: 词频饱和参数（1.2~2.0）
            b: 文档长度归一化（0=无归一化, 1=完全归一化）
            title_lines: 标题区域行数（前 N 行视为标题）
            title_boost: 标题区域 token 权重倍数
        """
        self.k1 = k1
        self.b = b
        self.title_lines = title_lines
        self.title_boost = title_boost
        self._doc_tokens: List[List[str]] = []   # 每篇文档的 token 列表
        self._doc_lengths: List[int] = []         # 每篇文档的 token 数量
        self._doc_lines: List[str] = []           # 每篇文档的原始文本（用于标题区域检测）
        self._avg_doc_len: float = 0.0
        self._df: Dict[str, int] = {}             # document frequency
        self._idf: Dict[str, float] = {}          # IDF cache
        self._num_docs: int = 0

    def index(self, documents: List[str]) -> None:
        """索引文档集合，保存原始文本用于标题区域评分"""
        self._doc_tokens = []
        self._doc_lengths = []
        self._doc_lines = []
        self._df = defaultdict(int)

        for doc in documents:
            tokens = Tokenizer.tokenize(doc)
            self._doc_tokens.append(tokens)
            self._doc_lengths.append(len(tokens))
            self._doc_lines.append(doc)
            for token in set(tokens):
                self._df[token] += 1

        self._num_docs = len(documents)
        self._avg_doc_len = (sum(self._doc_lengths) / self._num_docs) if self._num_docs > 0 else 0

        # 预计算 IDF
        self._idf = {}
        for token, df in self._df.items():
            self._idf[token] = math.log(1 + (self._num_docs - df + 0.5) / (df + 0.5))

    def search(self, query: str, top_k: int = 10,
               expand_synonyms: bool = False,
               phrase_boost: float = 3.0) -> List[Tuple[int, float]]:
        """
        搜索并返回 (doc_index, score) 列表

        Args:
            query: 查询字符串
            top_k: 返回前 k 个结果
            expand_synonyms: 是否启用同义词扩展
            phrase_boost: 短语精确匹配加分系数（查询词作为完整子串出现时的额外分数）
        """
        if not self._doc_tokens:
            return []

        query_tokens = Tokenizer.tokenize(query, precision=True)
        # 同义词扩展
        if expand_synonyms and len(query_tokens) <= 5:
            query_tokens = self._expand_query(query_tokens)

        # 预检查短语匹配（查询字符串作为完整子串出现在文档中）
        phrase_docs = set()
        query_lower = query.lower().strip()
        if len(query_lower) >= 3:
            for doc_idx, doc_text in enumerate(self._doc_lines):
                if query_lower in doc_text.lower():
                    phrase_docs.add(doc_idx)

        scores = []
        for doc_idx in range(self._num_docs):
            score = self._score(query_tokens, doc_idx)
            # 短语精确匹配加分
            if doc_idx in phrase_docs:
                phrase_bonus = phrase_boost * (1.0 + 0.1 * min(len(query_lower), 30))
                score += phrase_bonus
            if score > 0:
                scores.append((doc_idx, score))

        scores.sort(key=lambda x: -x[1])
        return scores[:top_k]

    def _score(self, query_tokens: List[str], doc_idx: int) -> float:
        """计算单个文档的 BM25 分数（含标题加权 + 精确匹配 boost）"""
        doc_tokens = self._doc_tokens[doc_idx]
        doc_len = self._doc_lengths[doc_idx]
        doc_text = self._doc_lines[doc_idx]

        # 标题区域 token 集合（前 title_lines 行）
        title_zone = self._get_title_zone(doc_text)
        title_tokens = set(Tokenizer.tokenize(title_zone)) if title_zone else set()

        # 词频统计
        tf = defaultdict(int)
        for t in doc_tokens:
            tf[t] += 1

        score = 0.0
        for token in query_tokens:
            if token not in tf:
                continue
            idf = self._idf.get(token, 0)
            term_freq = tf[token]
            numerator = term_freq * (self.k1 + 1)
            denominator = term_freq + self.k1 * (1 - self.b + self.b * doc_len / self._avg_doc_len)
            term_score = idf * numerator / denominator

            # 标题区域加权
            if token in title_tokens:
                term_score *= self.title_boost

            # 精确匹配 boost：完整标识符匹配（不分词）
            if self._is_identifier(token) and token in doc_text:
                term_score += 0.5

            score += term_score

        return score

    # ─── 内部辅助方法 ──────────────────────────────────

    def _get_title_zone(self, doc_text: str) -> str:
        """提取文档的标题区域（前 title_lines 行）"""
        lines = doc_text.split('\n', self.title_lines + 1)
        return '\n'.join(lines[:self.title_lines])

    @staticmethod
    def _is_identifier(token: str) -> bool:
        """判断 token 是否为完整标识符（非中文 n-gram 碎片）"""
        return bool(re.match(r'^[a-zA-Z_]\w{2,}$', token))

    @staticmethod
    def _expand_query(tokens: List[str]) -> List[str]:
        """同义词查询扩展（保留原始顺序 + 追加同义词）"""
        expanded = list(tokens)
        seen = set(t.lower() for t in tokens)
        for token in tokens:
            for syn in _EXPANDED_SYNONYMS.get(token.lower(), set()):
                if syn not in seen:
                    expanded.append(syn)
                    seen.add(syn)
        return expanded
